  <script>
    // 3섹션 카테고리 hover -> 4섹션 차트 제목 변경
    const buttons = document.querySelectorAll(".category-list__btn");
    const chartTitleBox = document.getElementById("chartTitleBox");
    const chartTitleA11y = document.getElementById("chartTitleA11y");

    const setActive = (btn) => {
      buttons.forEach(b => b.classList.remove("is-active"));
      if (btn) btn.classList.add("is-active");
    };

    const updateChartTitle = (category) => {
      const text = `차트 제목\\n(${category})`;
      chartTitleBox.textContent = text;
      chartTitleA11y.textContent = `차트 제목이 ${category}로 변경되었습니다.`;
    };

    buttons.forEach(btn => {
      btn.addEventListener("mouseenter", () => {
        setActive(btn);
        updateChartTitle(btn.dataset.category);
      });

      // 키보드 접근성(탭으로 이동 후 포커스 시에도 변경)
      btn.addEventListener("focus", () => {
        setActive(btn);
        updateChartTitle(btn.dataset.category);
      });
    });

    // 기본값(첫 카테고리)
    if (buttons[0]) {
      setActive(buttons[0]);
      updateChartTitle(buttons[0].dataset.category);
    }
  </script>
