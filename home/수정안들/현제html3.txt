<script>
  const buttons = document.querySelectorAll(".category-list__btn");
  const titleBox = document.getElementById("chartTitleBox");
  const titleA11y = document.getElementById("chartTitleA11y");

  const setActive = (btn) => {
    buttons.forEach(b => {
      b.classList.remove("is-active");
      b.setAttribute("aria-selected", "false");
    });
    btn.classList.add("is-active");
    btn.setAttribute("aria-selected", "true");
  };

  const updateTitle = (text) => {
    titleBox.textContent = `차트 제목 (${text})`;
    titleA11y.textContent = `${text} 차트로 변경됨`;
  };

  buttons.forEach((btn, index) => {
    const handler = () => {
      setActive(btn);
      updateTitle(btn.dataset.category);
    };

    btn.addEventListener("mouseenter", handler);
    btn.addEventListener("focus", handler);

    btn.addEventListener("keydown", (e) => {
      if (e.key === "ArrowRight") buttons[index + 1]?.focus();
      if (e.key === "ArrowLeft") buttons[index - 1]?.focus();
    });
  });

  // 기본값
  if (buttons[0]) {
    setActive(buttons[0]);
    updateTitle(buttons[0].dataset.category);
  }
</script>
